- 2. Script per trasformare Checkbox in Radio Button
Aggiungi questo codice nel tuo file functions.php o nel file principale del tuo plugin. Questo script cerca il box della tua tassonomia e trasforma tutti gli input checkbox in radio.

PHP

/**
 * Trasforma le checkbox della tassonomia Device Level in Radio Button
 */
add_action('admin_footer', function() {
    $screen = get_current_screen();
    
    // Esegui lo script solo nella pagina di modifica del tuo CPT
    // Sostituisci ALM_DEVICE_CPT_SLUG con la tua stringa se non è definita qui
    if ( $screen->post_type !== 'device' ) return; 

    ?>
    <script type="text/javascript">
        jQuery(document).ready(function($) {
            // ID del meta box: di solito è 'taxonomy-{slug}'
            // Sostituisci 'device_level' con il valore della tua costante ALM_DEVICE_LEVEL_TAXONOMY_SLUG
            var taxonomyName = 'device_level'; 
            var $container = $('#' + taxonomyName + 'div, #' + taxonomyName + 'checklist');

            $container.find('input[type="checkbox"]').each(function() {
                $(this).attr('type', 'radio');
                $(this).attr('name', 'tax_input[' + taxonomyName + '][]');
            });
        });
    </script>
    <?php
});
